MainPrgm StatsProg ;
Var
    let i , a , b , c : Int ;
    let sum , avg : Float ;
    let nums[10] : Int ;
    let vals[5] , ratios[5] : Float ;
    @define Const Pi    : Float = 3.1416 ;
    @define Const Limit : Int   = 10 ;

BeginPg
{
    a := 1 ;
    b := 2 ;
    c := a + b * 3 ;

    {-- Fill nums array --}
    for i from 0 to 9 step 1 {
        nums[i] := i + 1 ;
    }

    sum := 0 ;
    for i from 0 to 9 step 1 {
        sum := sum + nums[i] ;
    }

    if (Limit != 0) then {
        avg := sum / Limit ;
    } else {
        avg := 0 ;
    }

    output("Average: ") ;
    output(avg) ;

    for i from 0 to 4 step 1 {
        vals[i]   := (i + 1) * 2.5 ;
        ratios[i] := vals[i] / Pi ;
    }

    {-- display arrays --}
    for i from 0 to 4 step 1 {
        output(vals[i]) ;
        output(ratios[i]) ;
    }

    do {
        a := a + 1 ;
    } while (a < 5) ;
}
EndPg ;
