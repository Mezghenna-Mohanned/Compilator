MainPrgm SignedLitDemo ;
Var
    let a , b : Int ;
    let x , y : Float ;
BeginPg
{
    a := (-5) ;
    b := (+12) ;
    x := (-3.14) ;
    y := (+2.718) ;

    output("Int a = ") ;   output(a) ;
    output("Int b = ") ;   output(b) ;
    output("Float x = ") ; output(x) ;
    output("Float y = ") ; output(y) ;
}
EndPg ;
